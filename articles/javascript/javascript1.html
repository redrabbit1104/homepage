         //合計金額を計算する関数
         function calculate(){  

         //getElementByIdでpen要素を取得。選択したオプションに200を掛け変数penに代入。
         var pen = document.getElementById("pen").selectedIndex * 200;   

         //getElementByIdでpen要素を取得。選択したオプションに100を掛け変数penに代入。
         var note = document.getElementById("note").selectedIndex * 100;     
         
         //getElementByIdでoutput要素を取得。変数penとnoteの合計をoutput要素の値に代入。
         document.getElementById("output").value = pen + note;}   
                
         </script>
